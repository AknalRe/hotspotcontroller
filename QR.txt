const fs = require('fs');
const qr = require('qrcode');

function generateWiFiQRCode(ssid, password) {
    // const wifiUri = `WIFI:T:WPA;S:${ssid};P:${password};;`;
    const wifiUri = `http://haicantik.login/login?username=HaiCantikTamu%26password=HaiCantikTamu`;

    // Menyimpan QR Code sebagai file gambar PNG dengan resolusi lebih besar
    qr.toFile(
        `${ssid}_QRCode.png`,
        wifiUri,
        { errorCorrectionLevel: 'H', scale: 8 }, // Mengatur resolusi dengan opsi scale
        function (err) {
            if (err) throw err;
            console.log(`QR Code berhasil disimpan sebagai ${ssid}_QRCode.png`);
        }
    );
}
